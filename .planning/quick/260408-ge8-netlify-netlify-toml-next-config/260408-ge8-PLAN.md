```toml
# Netlify 배포 설정
# DataBase Backup Manager — Web Edition
#
# [중요] Socket.io 제한 사항:
# Netlify는 서버리스 함수 기반으로 동작하므로 영구적인 WebSocket 연결을 지원하지 않습니다.
# Socket.io를 사용하는 실시간 백업 진행 상황 기능은 Netlify에서 동작하지 않습니다.
# 실시간 기능이 필요한 경우 Railway, Render, Fly.io 등 장기 실행 서버 환경을 사용하세요.

[build]
  command = "npx prisma generate && next build"
  publish = ".next"

[build.environment]
  NODE_VERSION = "20"
  # 아래 환경변수는 Netlify 대시보드 > Site settings > Environment variables 에서 설정하세요:
  # DATABASE_URL          — PostgreSQL 연결 문자열 (예: postgresql://user:pass@host:5432/db)
  # NEXTAUTH_URL          — 배포된 사이트 URL (예: https://your-site.netlify.app)
  # NEXTAUTH_SECRET       — NextAuth 서명 키 (openssl rand -base64 32 로 생성)
  # ENCRYPTION_KEY        — AES-256 암호화 키 32바이트 hex (openssl rand -hex 32 로 생성)
  # AWS_ACCESS_KEY_ID     — S3 클라우드 스토리지 사용 시
  # AWS_SECRET_ACCESS_KEY — S3 클라우드 스토리지 사용 시
  # AWS_REGION            — S3 리전 (예: ap-northeast-2)

[[plugins]]
  package = "@netlify/plugin-nextjs"

# Next.js API Routes 및 서버 컴포넌트는 Netlify Functions으로 자동 변환됩니다.
# 별도의 redirects 설정 없이 @netlify/plugin-nextjs가 처리합니다.
```

주의: `publish = ".next"`는 `@netlify/plugin-nextjs` 플러그인 사용 시 올바른 값이다. `out`이나 `public`이 아님.
